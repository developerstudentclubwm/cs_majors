\title{Projectile Motion with Varying Air Resistance Strength}
\author{Luke Mrini}
\date{\today}
\maketitle

\begin{abstract}
    In this paper, we first investigate and visualize the gravitational potential generated by a point mass. Then we consider the effects of varying strengths of air resistance on a massive object undergoing projectile motion.
\end{abstract}

\section{Gravitational Potential}
For a point mass with mass $M$ the gravitational potential looks like
\begin{equation} \label{eq:gravity}
    \phi(r) = \frac{GM}{r}.
\end{equation}
See FIG. \ref{fig:gravity} for the potential.
\begin{figure}
        \includegraphics[width=1.0\columnwidth]{Gravitational potential.png}
        \centering
        \caption{
                \label{fig:gravity} 
                The gravitational potential for a point mass at the origin.
        }
\end{figure}
For more information see \cite{Triqueneaux:2020mrm}.
